24 EventSet & IndexBuild(EventSet &es, int B, int U) {
25  Event one;
26  string date;
27  date = "2021-03-01 10:32:00 UTC";
28  for (int i = 0; i < B * U; i++) {
29  date[8] = STR(1 + NBRAND(i, B, U))[0];
30  date[9] = STR(1 + NBRAND(i, B, U))[1];
31  one.setDateTime(date);
32  one.setBrand(BRAND(i, B, NUSERS));
33  one.setPrice(1 + i % B);
34  one.setUserID(USER(i, B, U));
35  one.setProductID((string) "Product_" + std::to_string(i % B) + "_" + std::to_string(i / B)); // Each brand has exactly 2 products
36  one.setSession((string) "Session_" + std::to_string(i / U) + "_" + std::to_string(i));
37  one.setType(VALID_TYPES[(i / ((B * U) / 4)) % 4]);
38  es.add(one);
39  }
40  return es;
41 }
