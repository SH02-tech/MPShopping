100  if (_nEntries == _capacity)

103  int successful = 1;

105  int pos = upper_bound(pair.getKey());
106 
107  for (int i=lower_bound(pair.getKey()); i<pos && successful == 1; i++)
108  if (_entries[i].to_string().compare(pair.to_string()) == 0)
109  successful = 0;
