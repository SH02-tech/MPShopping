167 Matrix PageRank(const Matrix & M, int max_iter, double threshold, int n){
168  // Define matrix Pr and set to the initial values
169  int nColumns = M.columns();
170 
171  string colLabels[nColumns];
172  string rowLabels[1] = {"Brands"};
173  M.getColLabels(colLabels);
174 
175  Matrix Pr(rowLabels, 1, colLabels, nColumns);
176 
177  for (int j=0; j<nColumns; j++) {
178  int brand_col = M.colIndexOf(colLabels[j]);
179  Pr(0, j) = M(M.rows()-1, brand_col);
180  }
181 
182  bool stable = false;
183 
184  // Iterate until the max number of iteration (MAX_ITER) is reached or the
185  // convergency value is less than the threshold
186 
187  for (int i=0; i<max_iter && !stable; i++) {
188  Matrix newPr = Pr * M;
189 
190  // We find the sum of all the differences, in absolute value. (1-norm)
191 
192  double sum = 0.0;
193 
194  for (int j=0; j<nColumns; j++)
195  sum += abs(newPr(0,j) - Pr(0,j));
196 
197  if (sum < threshold)
198  stable = true;
199 
200  Pr = newPr;
201 
202  cout << "PageRank at iteration " << i+1 << endl << Pr;
203  }
204 
205  // return the PageRank
206  return Pr;
207 }

220 Matrix computePageRank(Matrix & M, int max_iter, double threshold, int n = 10) {
221  Matrix pagerank;
222  double alfa = ALFA; // alfa value for teletransportation
223 
224  // Normalize the M matrix (all the rows have to sum one)
225  M.normalize();
226 
227  // Do teletransportation using the ALFA value
228 
229  M = alfa * M + ((1-alfa)/M.columns());
230 
231  // compute PageRank
232 
233  pagerank = PageRank(M,max_iter,threshold,n);
234 
235  return pagerank;
236 }
